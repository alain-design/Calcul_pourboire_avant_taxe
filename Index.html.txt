  <button onclick="calculateTip(0.15)">15% Tip</button>
  <button onclick="calculateTip(0.18)">18% Tip</button>
  <button onclick="calculateTip(0.20)">20% Tip</button>

  <div class="results" id="results"></div>

  <script>
    function calculateTip(tipRate) {
      const beforeTax = parseFloat(document.getElementById('amount').value);
      const taxRate = 0.14975;

      if (isNaN(beforeTax)) {
        alert("Entrez un montant valide.");
        return;
      }

      const tax = beforeTax * taxRate;
      const totalWithTax = beforeTax + tax;
      const tip = beforeTax * tipRate;
      const grandTotal = totalWithTax + tip;

      document.getElementById('results').innerHTML = `
        <p>Tip (${(tipRate * 100).toFixed(0)}%): $${tip.toFixed(2)}</p>
        <p>Total avec taxes + pourboire: $${grandTotal.toFixed(2)}</p>
      `;
    }
  </script>
